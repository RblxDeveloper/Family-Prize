rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isSignedIn() {
      return request.auth != null;
    }
    
    // Helper function to get user data (safe read for rule checks)
    function getUserData() {
      return exists(/databases/$(database)/documents/users/$(request.auth.uid))
        ? get(/databases/$(database)/documents/users/$(request.auth.uid)).data
        : null;
    }
    
    // Helper function to check if user is a parent
    function isParent() {
      return isSignedIn() && getUserData() != null && getUserData().role == 'parent';
    }

    // Helper function to check if user is a child
    function isChild() {
      return isSignedIn() && getUserData() != null && getUserData().role == 'child';
    }

    // Helper function to check if a document/family code belongs to the requester's family
    function isSameFamily(familyCode) {
      return isSignedIn() && getUserData() != null && getUserData().familyCode == familyCode;
    }
    
    // Users collection - MOST PERMISSIVE FOR SIGNUP
    match /users/{userId} {
      // Allow ANY authenticated user to create their own profile (for signup)
      allow create: if isSignedIn() && request.auth.uid == userId;

      // Read: owner or family members
      allow get: if isSignedIn() && (request.auth.uid == userId || isSameFamily(resource.data.familyCode));
      allow list: if isSignedIn() && isParent();

      // Update: owner or parent
      allow update: if (isSignedIn() && request.auth.uid == userId) || (isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode && resource.data.role == 'child');

      // Delete: not allowed
      allow delete: if false;
    }
    
    // Task Templates collection
    match /taskTemplates/{taskId} {
      allow read: if isSignedIn() && isSameFamily(resource.data.familyCode);
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow update, delete: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Rewards collection
    match /rewards/{rewardId} {
      allow read: if isSignedIn() && isSameFamily(resource.data.familyCode);
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow update, delete: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Submissions collection
    match /submissions/{submissionId} {
      // Allow ANY authenticated user to create submissions (checking role happens after signup)
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;

      // Read: user's own or family members
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isSameFamily(resource.data.familyCode)
      );

      // Parents can update (approve/decline)
      allow update: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Redeemed Rewards collection
    match /redeemedRewards/{redeemId} {
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow read: if isSignedIn() && (resource.data.userId == request.auth.uid || isSameFamily(resource.data.familyCode));
    }
    
    // Notifications collection
    match /notifications/{notificationId} {
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow read: if isSignedIn() && (resource.data.userId == request.auth.uid || isSameFamily(resource.data.familyCode));
      allow update: if isSignedIn() && resource.data.userId == request.auth.uid;
    }
  }
}
