rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isSignedIn() {
      return request.auth != null;
    }
    
    // Helper function to get user data (safe read for rule checks)
    // Use exists() guard so rules don't fail when a user profile hasn't been created yet.
    function getUserData() {
      return exists(/databases/$(database)/documents/users/$(request.auth.uid))
        ? get(/databases/$(database)/documents/users/$(request.auth.uid)).data
        : null;
    }
    
    // Helper function to check if user is a parent
    function isParent() {
      return isSignedIn() && getUserData() != null && getUserData().role == 'parent';
    }

    // Helper function to check if user is a child
    function isChild() {
      return isSignedIn() && getUserData() != null && getUserData().role == 'child';
    }

    // Helper function to check if a document/family code belongs to the requester's family
    function isSameFamily(familyCode) {
      return isSignedIn() && getUserData() != null && getUserData().familyCode == familyCode;
    }
    
    // Users collection
    match /users/{userId} {
      // Create: authenticated users can create their own profile document
      allow create: if isSignedIn() && request.auth.uid == userId;

      // Read: allow owners to read their profile; allow parents/children to read users in their family;
      // allow public reads for parent profiles (single-doc 'get' only) to assist family-code lookup.
      allow get: if isSignedIn() && (request.auth.uid == userId || resource.data.role == 'parent' || isSameFamily(resource.data.familyCode) || isParent());
      allow list: if isSignedIn() && isParent();

      // Update: users can update their own profile; parents can update child profiles within their family
      allow update: if (isSignedIn() && request.auth.uid == userId) || (isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode && resource.data.role == 'child');

      // Deleting user docs is disallowed by default
      allow delete: if false;
    }
    
    // Task Templates collection
    match /taskTemplates/{taskId} {
      // Family members can read task templates for their family
      allow read: if isSignedIn() && isSameFamily(resource.data.familyCode);

      // Only parents in the same family can create/update/delete
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow update, delete: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Rewards collection
    match /rewards/{rewardId} {
      allow read: if isSignedIn() && isSameFamily(resource.data.familyCode);
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow update, delete: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Submissions collection
    match /submissions/{submissionId} {
      // Children can create submissions for themselves in their family
      allow create: if isSignedIn() && isChild() &&
                      request.resource.data.userId == request.auth.uid &&
                      request.resource.data.familyCode == getUserData().familyCode;

      // Read: children can read their own submissions; parents and children in same family can read submissions for the family
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        (isSignedIn() && getUserData() != null && resource.data.familyCode == getUserData().familyCode)
      );

      // Parents can update (approve/decline) submissions in their family
      allow update: if isSignedIn() && isParent() && resource.data.familyCode == getUserData().familyCode;
    }
    
    // Redeemed Rewards collection
    match /redeemedRewards/{redeemId} {
      allow create: if isSignedIn() && isChild() && request.resource.data.userId == request.auth.uid && request.resource.data.familyCode == getUserData().familyCode;
      allow read: if isSignedIn() && (resource.data.userId == request.auth.uid || (getUserData() != null && resource.data.familyCode == getUserData().familyCode));
    }
    
    // Notifications collection
    match /notifications/{notificationId} {
      allow create: if isSignedIn() && isParent() && request.resource.data.familyCode == getUserData().familyCode;
      allow read: if isSignedIn() && (resource.data.userId == request.auth.uid || (getUserData() != null && resource.data.familyCode == getUserData().familyCode));
      allow update: if isSignedIn() && resource.data.userId == request.auth.uid;
    }
  }
}
